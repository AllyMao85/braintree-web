hosted-fields/external/index.js

'use strict';

var HostedFields = require('./hosted-fields');
var packageVersion = require('package.version');

/** @module braintree/hosted-fields */

/**
 * Fields used in {@link module:braintree/hosted-fields~createConfiguration createConfiguration}
 * @typedef {Object} field
 * @property {string} selector A CSS selector to find the container to place the hosted field into
 * @property {string} [placeholder] Will be used as the <code>placeholder=""</code> attribute of the input if <code>placeholder</code> is natively supported by the browser.
 */

module.exports = {
  /**
   * @static
   * @function create
   * @param {object} options Creation options
   * @param {client} options.client A {@link Client} instance
   * @param {object} [options.number] A {@link module:braintree/hosted-fields~field field} for card number
   * @param {object} [options.expirationDate] A {@link module:braintree/hosted-fields~field field} for expiration date in <code>MM/YYYY</code> format
   * @param {object} [options.expirationMonth] A {@link module:braintree/hosted-fields~field field} for expiration month in <code>MM</code> format
   * @param {object} [options.expirationYear] A {@link module:braintree/hosted-fields~field field} for expiration year in <code>YYYY</code> format
   * @param {object} [options.cvv] A {@link module:braintree/hosted-fields~field field} for 3 or 4 digit CVV or CID
   * @param {object} [options.postalCode] A {@link module:braintree/hosted-fields~field field} for postal or region code
   * @param {errback} callback The second argument, <code>data</code>, is the {@link HostedFields} instance
   * @returns {void}
   */
  create: function (options, callback) {
    var integration;

    try {
      integration = new HostedFields(options);
    } catch (err) {
      callback(err);
      return;
    }

    integration.on('ready', function () {
      callback(null, integration);
    });
  },
  /** @property {string} VERSION the current version of the SDK */
  VERSION: packageVersion
};
